        function toggleMusic() {

            if (audio.paused) {

                audio.play();

                musicIconOn.classList.remove('hidden');

                musicIconOff.classList.add('hidden');

            } else {

                audio.pause();

                musicIconOn.classList.add('hidden');

                musicIconOff.classList.remove('hidden');

            }

        }


        musicToggle.addEventListener('click', toggleMusic);


        function updateDots() {

            dotsContainer.innerHTML = '';

            story.pages.forEach((_, i) => {

                const dot = document.createElement('div');

                dot.className = `h-1.5 rounded-full transition-all duration-500 ${i === currentPage ? 'w-8 bg-amber-600' : 'w-2 bg-stone-200'}`;

                dotsContainer.appendChild(dot);

            });

        }


        function updatePage() {

            const page = story.pages[currentPage];

            subtitleEl.innerText = page.subtitle;

            chapterTitleEl.innerText = page.chapterTitle;

            contentEl.innerText = page.content;

            

            imgWrapper.innerHTML = '';

            const img = document.createElement('img');

            img.className = "main-img fade-in";

            img.alt = "Ilustración de " + page.chapterTitle;

            img.src = page.imageUrl;

            

            imgWrapper.appendChild(img);

            prevBtn.disabled = currentPage === 0;

            

            if (currentPage === story.pages.length - 1) {

                nextBtn.innerHTML = '<span>Comenzar de nuevo</span>';

            } else {

                nextBtn.innerHTML = 'Siguiente <svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="m9 18 6-6-6-6"/></svg>';

            contentArea.classList.add('fade-in');

            updateDots();

        }


        function handleNavigation() {

            if (!musicStarted) {

                audio.play().catch(() => {

                    console.log("Audio waiting for interaction");

                });

                musicStarted = true;

            }

        }


        prevBtn.addEventListener('click', () => {

            if (currentPage > 0) { 

                currentPage--; 

                updatePage(); 

                window.scrollTo({top: 0, behavior: 'smooth'}); 

            }

        });


        nextBtn.addEventListener('click', () => {

            handleNavigation();

            if (currentPage < story.pages.length - 1) {

                currentPage++;

            } else {

                currentPage = 0;

            }

            updatePage(); 

            window.scrollTo({top: 0, behavior: 'smooth'});

        });
